# Retro 2.0 Plan 4: Projection Pipeline Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire projection into the v2 pipeline so approved knowledge nodes get written to files, PRs get created for shared changes, and PR state is tracked automatically.

**Architecture:** Three additions to the existing v2 pipeline: (1) dash immediate projection for global rules to `~/.claude/CLAUDE.md`, (2) `retro run` projection step that writes files and creates batched PRs per project, (3) `retro run` sync step that checks PR state via `gh`. Reuses existing v1 projection functions (`update_claude_md_content`, `generate_with_retry`) and git helpers (`create_branch`, `commit_files`, `create_pr`). A `node_to_pattern()` adapter bridges v2 nodes to v1 skill generation.

**Tech Stack:** Rust, rusqlite (bundled), serde, serde_json, chrono

**Spec:** `docs/superpowers/specs/2026-03-24-retro-2.0-plan-4-projection-pipeline-design.md`
**Depends on:** Plans 1–3 complete on `feature/retro-2.0-surfaces`

use crate::errors::CoreError;
use std::path::Path;

/// Append a single rule to a CLAUDE.md file's managed section.
/// Creates the file and managed section if missing.
pub fn project_rule_to_claude_md(path: &Path, rule: &str) -> Result<(), CoreError> {
    let existing = if path.exists() {
        std::fs::read_to_string(path)
            .map_err(|e| CoreError::Io(format!("reading CLAUDE.md: {e}")))?
    } else {
        String::new()
    };

    // Read existing rules, add the new one
    let mut rules: Vec<String> = read_managed_section(&existing).unwrap_or_default();
    if !rules.iter().any(|r| r == rule) {
        rules.push(rule.to_string());
    }

    let updated = update_claude_md_content(&existing, &rules);

    if let Some(parent) = path.parent() {
        std::fs::create_dir_all(parent)
            .map_err(|e| CoreError::Io(format!("creating directory: {e}")))?;
    }
    std::fs::write(path, &updated)
        .map_err(|e| CoreError::Io(format!("writing CLAUDE.md: {e}")))?;

    Ok(())
}

use crate::models::{KnowledgeNode, NodeType, Pattern, PatternType, PatternStatus, SuggestedTarget};

/// Convert a v2 KnowledgeNode to a v1 Pattern for skill generation.
pub fn node_to_pattern(node: &KnowledgeNode) -> Pattern {
    let pattern_type = match node.node_type {
        NodeType::Skill => PatternType::WorkflowPattern,
        NodeType::Rule | NodeType::Directive => PatternType::RepetitiveInstruction,
        NodeType::Pattern => PatternType::RecurringMistake,
        NodeType::Preference | NodeType::Memory => PatternType::WorkflowPattern,
    };
    let suggested_target = match node.node_type {
        NodeType::Skill => SuggestedTarget::Skill,
        NodeType::Rule | NodeType::Directive | NodeType::Pattern
        | NodeType::Preference | NodeType::Memory => SuggestedTarget::ClaudeMd,
    };

    Pattern {
        id: node.id.clone(),
        pattern_type,
        description: node.content.clone(),
        confidence: node.confidence,
        times_seen: 1,
        first_seen: node.created_at,
        last_seen: node.updated_at,
        last_projected: None,
        status: PatternStatus::Active,
        source_sessions: vec![],
        related_files: vec![],
        suggested_content: node.content.clone(),
        suggested_target,
        project: node.project_id.clone(),
        generation_failed: false,
    }
}

```rust
/// Create a retro PR: chdir → stash → branch from default → write files → commit → push → PR → restore.
/// Returns the PR URL on success, or None if PR creation was skipped.
/// IMPORTANT: sets current directory to project_path for git commands, then restores.
pub fn create_retro_pr(
    project_path: &str,
    files: &[(&str, &str)],  // (relative_path, content) pairs
    commit_message: &str,
    pr_title: &str,
    pr_body: &str,
) -> Result<Option<String>, CoreError> {
    // Save and change working directory (git commands use cwd)
    let original_dir = std::env::current_dir()
        .map_err(|e| CoreError::Io(format!("getting cwd: {e}")))?;
    std::env::set_current_dir(project_path)
        .map_err(|e| CoreError::Io(format!("changing to {project_path}: {e}")))?;

    let result = create_retro_pr_inner(project_path, files, commit_message, pr_title, pr_body);

    // Always restore original directory
    let _ = std::env::set_current_dir(&original_dir);

    result
}

fn create_retro_pr_inner(
    project_path: &str,
    files: &[(&str, &str)],
    commit_message: &str,
    pr_title: &str,
    pr_body: &str,
) -> Result<Option<String>, CoreError> {
    // Save current state
    let original_branch = current_branch()?;
    let default = default_branch()?;
    let _ = fetch_branch(&default);

    // Stash
    let stashed = stash_push()?;

    // Create branch
    let timestamp = chrono::Utc::now().format("%Y%m%d-%H%M%S");
    let branch_name = format!("retro/updates-{timestamp}");
    if let Err(e) = create_branch(&branch_name, Some(&format!("origin/{default}"))) {
        if stashed { let _ = stash_pop(); }
        return Err(e);
    }

    // Write files
    for (path, content) in files {
        let full_path = std::path::Path::new(project_path).join(path);
        if let Some(parent) = full_path.parent() {
            let _ = std::fs::create_dir_all(parent);
        }
        std::fs::write(&full_path, content)
            .map_err(|e| CoreError::Io(format!("writing {path}: {e}")))?;
    }

    // Commit
    let file_paths: Vec<&str> = files.iter().map(|(p, _)| *p).collect();
    commit_files(&file_paths, commit_message)?;

    // Push
    let pr_url = match push_current_branch() {
        Ok(()) => {
            if is_gh_available() {
                match create_pr(pr_title, pr_body, &default) {
                    Ok(url) => Some(url),
                    Err(_) => None,
                }
            } else {
                None
            }
        }
        Err(_) => None,
    };

    // Restore branch
    let _ = checkout_branch(&original_branch);
    if stashed { let _ = stash_pop(); }

    Ok(pr_url)
}
```

let unprojected = db::get_unprojected_nodes(conn)?;

    println!("  {} no unprojected nodes", "Skipping:".dimmed());

    let mut projected_count = 0usize;
    let mut pr_url: Option<String> = None;

    // Group by scope
    let global_rules: Vec<&KnowledgeNode> = unprojected.iter()
        .filter(|n| n.scope == NodeScope::Global && n.project_id.is_none())
        .filter(|n| matches!(n.node_type, NodeType::Rule | NodeType::Directive | NodeType::Preference))
        .collect();
    let global_skills: Vec<&KnowledgeNode> = unprojected.iter()
        .filter(|n| n.scope == NodeScope::Global && matches!(n.node_type, NodeType::Skill))
        .collect();
    let project_nodes: Vec<&KnowledgeNode> = unprojected.iter()
        .filter(|n| n.scope == NodeScope::Project)
        .collect();

    // Global rules → ~/.claude/CLAUDE.md
    if !global_rules.is_empty() {
        let home = std::env::var("HOME").unwrap_or_else(|_| "/tmp".to_string());
        let claude_md_path = std::path::PathBuf::from(&home).join(".claude").join("CLAUDE.md");
        for node in &global_rules {
            if retro_core::projection::claude_md::project_rule_to_claude_md(&claude_md_path, &node.content).is_ok() {
                db::mark_node_projected(conn, &node.id)?;
                projected_count += 1;
            }
        }
        if verbose {
            println!("  {} {} global rules to ~/.claude/CLAUDE.md", "Projected".green(), global_rules.len());
        }

    // Global skills → ~/.claude/skills/ (AI call)
    // TODO: implement when AI budget permits — skip for now
    for node in &global_skills {
        if verbose {
            println!("  {} skill generation deferred: {}", "Skipped".dimmed(), retro_core::util::truncate_str(&node.content, 60));
        }
    }

    // Project-scoped nodes → PR
    if !project_nodes.is_empty() {
        // Collect rules as strings for CLAUDE.md
        let rules: Vec<String> = project_nodes.iter()
            .filter(|n| matches!(n.node_type, NodeType::Rule | NodeType::Directive | NodeType::Preference | NodeType::Pattern))
            .map(|n| n.content.clone())
            .collect();

        if !rules.is_empty() {
            // Build CLAUDE.md content
            let claude_md_path = format!("{}/CLAUDE.md", project_path);
            let existing = std::fs::read_to_string(&claude_md_path).unwrap_or_default();
            let updated = retro_core::projection::claude_md::update_claude_md_content(&existing, &rules);

            // Create PR
            match retro_core::git::create_retro_pr(
                project_path,
                &[("CLAUDE.md", &updated)],
                "retro: update CLAUDE.md with discovered rules",
                "retro: update CLAUDE.md rules",
                &format!("Retro discovered {} rule(s) from your sessions.\n\nApproved via `retro dash`.", rules.len()),

                Ok(Some(url)) => {
                    for node in &project_nodes {
                        db::mark_node_projected_with_pr(conn, &node.id, &url)?;
                    }
                    pr_url = Some(url.clone());
                    projected_count += project_nodes.len();
                    println!("  {} PR: {}", "Created".green(), url.cyan());

                Ok(None) => {

                }
                Err(e) => {
                    eprintln!("  {} PR creation: {e}", "Error".red());
                }
            }
        }
    }

println!("{}", "Step 5/6: Syncing PR state...".cyan());

let nodes_with_pr = db::get_nodes_with_pr(conn)?;
if nodes_with_pr.is_empty() {
    if verbose {
        println!("  {} no open PRs to check", "Skipping:".dimmed());
    }
} else if dry_run {
    let unique_prs: std::collections::HashSet<&str> = nodes_with_pr.iter()
        .filter_map(|n| n.pr_url.as_deref())
        .collect();
    println!("  {} {} PRs to check", "[dry-run]".yellow(), unique_prs.len());
} else {
    let unique_prs: std::collections::HashSet<String> = nodes_with_pr.iter()
        .filter_map(|n| n.pr_url.clone())
        .collect();

    for pr_url in &unique_prs {
        match retro_core::git::pr_state(pr_url) {
            Ok(state) => {
                match state.as_str() {
                    "MERGED" => {
                        db::clear_node_pr(conn, pr_url)?;
                        if verbose {
                            println!("  {} PR merged: {}", "Cleared".green(), pr_url);
                        }
                    }
                    "CLOSED" => {
                        db::dismiss_nodes_for_pr(conn, pr_url)?;
                        println!("  {} PR closed — nodes dismissed: {}", "Dismissed".yellow(), pr_url);
                    }
                    _ => {
                        if verbose {
                            println!("  {} PR open: {}", "Waiting".dimmed(), pr_url);
                        }
                    }
                }
            }
            Err(e) => {
                if verbose {
                    eprintln!("  {} checking PR {}: {e}", "Warning".yellow(), pr_url);
                }
            }
        }
    }
}
